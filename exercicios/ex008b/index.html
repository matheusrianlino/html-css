users = list()
user = dict()
c = ''
while True:
    user['name'] = input('Digite o nome: ')
    user['idade'] = input('Digite a idade: ')
    users.append(user.copy())
    while c != 'S' and c != 'N':
        c = str(input('Deseja continuar? [S/N]: ')).strip().upper()
    if c == 'N':
        break
    c = ''
for c in range(0, len(users)):
    print(f'Nome: {users[c]["name"]}')
    print(f'Idade: {users[c]["idade"]}')
